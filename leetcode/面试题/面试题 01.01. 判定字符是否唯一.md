# 实现一个算法，确定一个字符串 s 的所有字符是否全都不同。

示例 1：

输入: s = "leetcode"
输出: false 
示例 2：

输入: s = "abc"
输出: true
限制：

0 <= len(s) <= 100
如果你不使用额外的数据结构，会很加分。

GO:
```
func isUnique(astr string) bool {
length := len(astr)
    for i := 0; i < length -1; i ++ {
        for j := i + 1; j < length; j ++ {
            if astr[i] == astr[j] {
                return false
            }
        }
    }
    return true
}
```

PHP:
```
class Solution {

    /**
     * @param String $astr
     * @return Boolean
     */
    function isUnique($astr) {
        $len = strlen($astr);    
        for ($i = 0; $i < $len -1; $i ++) {
        for ($j = $i + 1; $j < $len; $j ++) {
            if ($astr[$i] == $astr[$j]) {
                return false;
            }
        }
    }
    return true;
    }
}
```
